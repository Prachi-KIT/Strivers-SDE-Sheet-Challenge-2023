#include <bits/stdc++.h>

pair<int,int> missingAndRepeating(vector<int> &arr, int n)
{
	int newArr[n+1]={0};
	for(int i=0;i<n;i++)
	{
		newArr[arr[i]]++;
	}
    int repeating=-1,missing=-1;
	for(int i=0;i<=n;i++)
	{
		if(newArr[i]==2)
		{
             repeating=i;
		}
		else if(newArr[i]==0)
		{
			missing=i;
		}

		if(repeating==-1 && missing==-1)
		{
			break;
		}
	}

	return {missing,repeating};
	
}
